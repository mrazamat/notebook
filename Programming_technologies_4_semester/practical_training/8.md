# Использование бесплатных API для перевода текста

## Введение в использование API

API (Application Programming Interface) — это набор правил и инструментов для взаимодействия между приложениями. В этой лабораторной работе мы будем работать с бесплатными API для машинного перевода: Google Translate и DeepL.

### Основные концепции

API перевода позволяют:
- Переводить текст между различными языками
- Интегрировать переводы в веб-приложения
- Автоматизировать обработку многоязычного контента

### Google Translate API

Google Translate предоставляет бесплатный перевод без ключа через веб-интерфейс или специальные библиотеки. Для простых проектов можно использовать библиотеку `google-translate-official` или работать с неофициальным API через библиотеку `google_translator_toolkit`.

### DeepL API

DeepL известен высоким качеством перевода. Бесплатный API доступен с ограничением в 500,000 символов в месяц. Требует регистрации и получения API ключа.

### Задача 1.2: Простой пример перевода текста

Создайте файл `simple_translation.py`:

```python
from google_translate_official import google_translate

def translate_with_google(text, source_lang='en', target_lang='ru'):
    """
    Переводит текст используя Google Translate
    
    Args:
        text: Текст для перевода
        source_lang: Исходный язык (по умолчанию английский)
        target_lang: Целевой язык (по умолчанию русский)
    
    Returns:
        Переведенный текст
    """
    try:
        result = google_translate(
            text=text,
            source_language=source_lang,
            target_language=target_lang
        )
        return result
    except Exception as e:
        print(f"Ошибка при переводе: {e}")
        return None

# Пример использования
if __name__ == "__main__":
    english_text = "Hello, this is a test message."
    translated = translate_with_google(english_text)
    print(f"Исходный текст: {english_text}")
    print(f"Переведенный текст: {translated}")
```

import deepl

def translate_with_deepl(text, target_lang='RU', source_lang=None, api_key=None):
    """
    Переводит текст используя DeepL API
    
    Args:
        text: Текст для перевода
        target_lang: Целевой язык (код языка DeepL)
        source_lang: Исходный язык (опционально)
        api_key: API ключ DeepL
    
    Returns:
        Переведенный текст
    """

## Часть 2: Работа с SQLite для хранения переводов

### Задача 2.1: Создание схемы базы данных

Создайте файл `database.py`:

```python
import sqlite3
from datetime import datetime

class TranslationDatabase:
    """
    Класс для управления базой данных переводов
    """
    
    def __init__(self, db_name='translations.db'):
        """
        Инициализирует подключение к базе данных
        
        Args:
            db_name: Имя файла базы данных
        """
        self.db_name = db_name
        self.create_tables()
    
    def create_tables(self):
        """
        Создает необходимые таблицы в базе данных
        """
        conn = sqlite3.connect(self.db_name)
        cursor = conn.cursor()
        
        # Таблица для хранения переводов
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS translations (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                source_text TEXT NOT NULL,
                translated_text TEXT NOT NULL,
                source_language TEXT DEFAULT 'en',
                target_language TEXT DEFAULT 'ru',
                api_service TEXT,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        ''')
        
        # Таблица для хранения языков
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS languages (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                code TEXT UNIQUE NOT NULL,
                name TEXT NOT NULL,
                native_name TEXT
            )
        ''')
        
        conn.commit()
        conn.close()
    
    def add_translation(self, source_text, translated_text, 
                       source_lang='en', target_lang='ru', api_service=None):
        """
        Добавляет перевод в базу данных
        
        Args:
            source_text: Исходный текст
            translated_text: Переведенный текст
            source_lang: Код исходного языка
            target_lang: Код целевого языка
            api_service: Название использованного API сервиса
        
        Returns:
            ID добавленной записи
        """
        conn = sqlite3.connect(self.db_name)
        cursor = conn.cursor()
        
        cursor.execute('''
            INSERT INTO translations 
            (source_text, translated_text, source_language, 
             target_language, api_service)
            VALUES (?, ?, ?, ?, ?)
        ''', (source_text, translated_text, source_lang, 
              target_lang, api_service))
        
        translation_id = cursor.lastrowid
        conn.commit()
        conn.close()
        
        return translation_id
    
    def get_translation(self, translation_id):
        """
        Получает перевод по ID
        
        Args:
            translation_id: ID перевода
        
        Returns:
            Словарь с данными перевода или None
        """
        conn = sqlite3.connect(self.db_name)
        cursor = conn.cursor()
        
        cursor.execute('SELECT * FROM translations WHERE id = ?', 
                      (translation_id,))
        row = cursor.fetchone()
        conn.close()
        
        if row:
            return {
                'id': row[0],
                'source_text': row[1],
                'translated_text': row[2],
                'source_language': row[3],
                'target_language': row[4],
                'api_service': row[5],
                'created_at': row[6]
            }
        return None
    
    def get_all_translations(self):
        """
        Получает все переводы из базы данных
        
        Returns:
            Список словарей с переводами
        """
        conn = sqlite3.connect(self.db_name)
        cursor = conn.cursor()
        
        cursor.execute('SELECT * FROM translations ORDER BY created_at DESC')
        rows = cursor.fetchall()
        conn.close()
        
        translations = []
        for row in rows:
            translations.append({
                'id': row[0],
                'source_text': row[1],
                'translated_text': row[2],
                'source_language': row[3],
                'target_language': row[4],
                'api_service': row[5],
                'created_at': row[6]
            })
        
        return translations
    
    def search_translations(self, query, search_in='source_text'):
        """
        Ищет переводы по запросу
        
        Args:
            query: Поисковый запрос
            search_in: В каком поле искать ('source_text' или 'translated_text')
        
        Returns:
            Список найденных переводов
        """
        conn = sqlite3.connect(self.db_name)
        cursor = conn.cursor()
        
        sql = f'SELECT * FROM translations WHERE {search_in} LIKE ?'
        cursor.execute(sql, (f'%{query}%',))
        rows = cursor.fetchall()
        conn.close()
        
        translations = []
        for row in rows:
            translations.append({
                'id': row[0],
                'source_text': row[1],
                'translated_text': row[2],
                'source_language': row[3],
                'target_language': row[4],
                'api_service': row[5],
                'created_at': row[6]
            })
        
        return translations
```

## Часть 3: Flask веб-приложение для работы с переводами

### Задача 3.1: Создание основного приложения Flask

Создайте файл `app.py`:

```python
from flask import Flask, render_template, request, jsonify
from database import TranslationDatabase
from google_translate_official import google_translate
import deepl

app = Flask(__name__)
db = TranslationDatabase('translations.db')

# Конфигурация
DEEPL_API_KEY = 'your_deepl_api_key_here'  # Замените на ваш API ключ
SUPPORTED_LANGUAGES = {
    'en': 'English',
    'ru': 'Russian',
    'de': 'German',
    'fr': 'French',
    'es': 'Spanish',
    'zh': 'Chinese'
}

@app.route('/')
def index():
    """
    Главная страница приложения
    """
    return render_template('index.html', languages=SUPPORTED_LANGUAGES)

@app.route('/api/translate', methods=['POST'])
def translate():
    """
    API endpoint для перевода текста
    """
    data = request.get_json()
    
    source_text = data.get('text', '').strip()
    source_lang = data.get('source_lang', 'en')
    target_lang = data.get('target_lang', 'ru')
    api_service = data.get('service', 'google')
    
    if not source_text:
        return jsonify({'error': 'Пустой текст'}), 400
    
    try:
        if api_service == 'google':
            translated_text = google_translate(
                text=source_text,
                source_language=source_lang,
                target_language=target_lang
            )
        elif api_service == 'deepl':
            translator = deepl.Translator(DEEPL_API_KEY)
            result = translator.translate_text(
                source_text,
                source_language=source_lang,
                target_language=target_lang
            )
            translated_text = result.text
        else:
            return jsonify({'error': 'Неподдерживаемый сервис'}), 400
        
        # Сохранение в базу данных
        translation_id = db.add_translation(
            source_text=source_text,
            translated_text=translated_text,
            source_lang=source_lang,
            target_lang=target_lang,
            api_service=api_service
        )
        
        return jsonify({
            'id': translation_id,
            'source_text': source_text,
            'translated_text': translated_text,
            'api_service': api_service,
            'status': 'success'
        })
    
    except Exception as e:
        return jsonify({'error': str(e)}), 500

@app.route('/api/history')
def get_history():
    """
    Получает историю переводов
    """
    translations = db.get_all_translations()
    return jsonify(translations)

@app.route('/api/search', methods=['GET'])
def search():
    """
    Поиск переводов
    """
    query = request.args.get('q', '').strip()
    search_in = request.args.get('search_in', 'source_text')
    
    if not query:
        return jsonify({'error': 'Пустой поисковый запрос'}), 400
    
    results = db.search_translations(query, search_in)
    return jsonify(results)

@app.route('/api/translation/<int:translation_id>')
def get_translation_details(translation_id):
    """
    Получает детали конкретного перевода
    """
    translation = db.get_translation(translation_id)
    
    if translation:
        return jsonify(translation)
    else:
        return jsonify({'error': 'Перевод не найден'}), 404

if __name__ == '__main__':
    app.run(debug=True)
```

### Упражнение 2: Работа с файлами

Создайте скрипт `translate_file.py` для перевода текста из файла:

```python
import os
from database import TranslationDatabase
from google_translate_official import google_translate

def translate_text_file(input_file, output_file, source_lang='en', 
                       target_lang='ru', api_service='google'):
    """
    Переводит текст из файла и сохраняет результат
    """
    db = TranslationDatabase('file_translations.db')
    
    # Чтение исходного файла
    if not os.path.exists(input_file):
        print(f"Файл {input_file} не найден!")
        return
    
    with open(input_file, 'r', encoding='utf-8') as f:
        lines = f.readlines()
    
    translated_lines = []
    
    # Перевод каждой строки
    for i, line in enumerate(lines, 1):
        line = line.strip()
        if line:  # Пропускаем пустые строки
            print(f"Перевод строки {i}: {line[:50]}...")
            translated = google_translate(line, source_lang, target_lang)
            translated_lines.append(translated + '\n')
            
            # Сохранение в базу данных
            db.add_translation(line, translated, source_lang, 
                             target_lang, api_service)
        else:
            translated_lines.append('\n')
    
    # Запись результата в файл
    with open(output_file, 'w', encoding='utf-8') as f:
        f.writelines(translated_lines)
    
    print(f"\nПереводы сохранены в {output_file}")
    print(f"Всего переведено строк: {len([l for l in lines if l.strip()])}")

if __name__ == "__main__":
    translate_text_file('input_text.txt', 'output_text.txt')
```

### Упражнение 4: Статистика переводов

Создайте скрипт `translation_statistics.py`:

```python
from database import TranslationDatabase
from collections import Counter

def get_translation_statistics():
    """
    Анализирует статистику переводов
    """
    db = TranslationDatabase('translations.db')
    translations = db.get_all_translations()
    
    if not translations:
        print("Нет переводов для анализа!")
        return
    
    # Подсчет по API сервисам
    services = [t['api_service'] for t in translations]
    service_counts = Counter(services)
    
    # Подсчет по языковым парам
    language_pairs = []
    for t in translations:
        pair = f"{t['source_language']} -> {t['target_language']}"
        language_pairs.append(pair)
    language_counts = Counter(language_pairs)
    
    # Подсчет средней длины текстов
    avg_source_length = sum(len(t['source_text']) 
                           for t in translations) / len(translations)
    avg_trans_length = sum(len(t['translated_text']) 
                          for t in translations) / len(translations)
    
    # Вывод статистики
    print("\n" + "="*60)
    print("СТАТИСТИКА ПЕРЕВОДОВ")
    print("="*60)
    print(f"\nВсего переводов: {len(translations)}")
    
    print("\nПо API сервисам:")
    for service, count in service_counts.items():
        print(f"  {service}: {count}")
    
    print("\nПо языковым парам:")
    for pair, count in language_counts.items():
        print(f"  {pair}: {count}")
    
    print(f"\nСредняя длина исходного текста: {avg_source_length:.1f} символов")
    print(f"Средняя длина переведенного текста: {avg_trans_length:.1f} символов")
    print("="*60)

if __name__ == "__main__":
    get_translation_statistics()
```
