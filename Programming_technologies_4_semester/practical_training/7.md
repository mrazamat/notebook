# Простая система перевода с Flask и SQLite

## Введение

В этой лабораторной работе вы создадите веб-приложение для управления переводами слов и фраз. Приложение будет использовать Flask как веб-фреймворк и SQLite для хранения данных о переводах между двумя языками (например, русский-английский).

Эта работа демонстрирует:
- Создание базы данных SQLite для хранения словаря переводов
- Разработку REST API с использованием Flask
- Реализацию функций поиска и добавления переводов
- Интеграцию веб-интерфейса с базой данных

## Теоретические основы

### Flask и веб-приложения

Flask — это микрофреймворк Python для создания веб-приложений. Он позволяет легко создавать HTTP маршруты (routes), обрабатывать запросы и возвращать ответы.

Основные компоненты:
- **Маршруты (Routes)**: URL пути, на которые приложение реагирует
- **Шаблоны (Templates)**: HTML файлы с динамическим содержимым
- **Статические файлы**: CSS, JavaScript, изображения

### SQLite для хранения переводов

SQLite — это встроенная база данных, которая хранит данные в одном файле. Для приложения переводов мы создадим таблицу со следующей структурой:

```
translation
├── id (PRIMARY KEY)
├── word_source (исходное слово)
├── language_source (язык источника)
├── word_target (переводное слово)
├── language_target (язык перевода)
└── timestamp (дата добавления)
```

### Поиск и фильтрация

Система должна поддерживать:
- Поиск слова по точному совпадению
- Получение всех переводов для одного слова
- Фильтрацию по направлению перевода (например, только русский → английский)

### Задача 1: Инициализация проекта и создание базы данных

Создайте файл `translation_app.py`, который инициализирует приложение и создаёт структуру базы данных:

```python
import sqlite3
import os
from datetime import datetime
from flask import Flask, render_template, request, jsonify

app = Flask(__name__)
DATABASE = 'translation.db'

def get_database():
    """Получить соединение с базой данных"""
    connection = sqlite3.connect(DATABASE)
    connection.row_factory = sqlite3.Row
    return connection

def init_database():
    """Инициализировать базу данных с таблицей translation"""
    if not os.path.exists(DATABASE):
        connection = get_database()
        cursor = connection.cursor()
        
        cursor.execute('''
            CREATE TABLE translation (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                word_source TEXT NOT NULL,
                language_source TEXT NOT NULL,
                word_target TEXT NOT NULL,
                language_target TEXT NOT NULL,
                timestamp DATETIME DEFAULT CURRENT_TIMESTAMP
            )
        ''')
        
        # Добавить примеры переводов
        sample_data = [
            ('hello', 'English', 'привет', 'Russian'),
            ('hello', 'English', 'здравствуйте', 'Russian'),
            ('cat', 'English', 'кот', 'Russian'),
            ('dog', 'English', 'собака', 'Russian'),
            ('book', 'English', 'книга', 'Russian'),
            ('sun', 'English', 'солнце', 'Russian'),
            ('tree', 'English', 'дерево', 'Russian'),
        ]
        
        cursor.executemany('''
            INSERT INTO translation (word_source, language_source, word_target, language_target)
            VALUES (?, ?, ?, ?)
        ''', sample_data)
        
        connection.commit()
        connection.close()
        print("База данных инициализирована успешно")

# Инициализировать базу при запуске
if __name__ == '__main__':
    init_database()
    app.run(debug=True)
```

@app.route('/api/translate', methods=['GET'])
def get_translate():
    """API для поиска перевода слова"""
    word = request.args.get('word', '').lower()
    source_lang = request.args.get('source_lang', 'English')
    target_lang = request.args.get('target_lang', 'Russian')
    
    if not word:
        return jsonify({'error': 'Слово не указано'}), 400
    
    connection = get_database()
    cursor = connection.cursor()
    
    cursor.execute('''
        SELECT word_target FROM translation
        WHERE LOWER(word_source) = ?
        AND language_source = ?
        AND language_target = ?
    ''', (word, source_lang, target_lang))
    
    results = cursor.fetchall()
    connection.close()
    
    if not results:
        return jsonify({
            'word': word,
            'translations': [],
            'found': False
        }), 404
    
    translations = [row[0] for row in results]
    return jsonify({
        'word': word,
        'translations': translations,
        'found': True
    })

@app.route('/api/add_translation', methods=['POST'])
def add_translation():
    """API для добавления нового перевода"""
    data = request.get_json()
    
    # Валидация входных данных
    required_fields = ['word_source', 'language_source', 'word_target', 'language_target']
    if not all(field in data for field in required_fields):
        return jsonify({'error': 'Отсутствуют обязательные поля'}), 400
    
    word_source = data.get('word_source', '').lower().strip()
    language_source = data.get('language_source', '').strip()
    word_target = data.get('word_target', '').lower().strip()
    language_target = data.get('language_target', '').strip()
    
    if not all([word_source, language_source, word_target, language_target]):
        return jsonify({'error': 'Поля не могут быть пусты'}), 400
    
    connection = get_database()
    cursor = connection.cursor()
    
    try:
        cursor.execute('''
            INSERT INTO translation (word_source, language_source, word_target, language_target)
            VALUES (?, ?, ?, ?)
        ''', (word_source, language_source, word_target, language_target))
        
        connection.commit()
        new_id = cursor.lastrowid
        connection.close()
        
        return jsonify({
            'message': 'Перевод успешно добавлен',
            'id': new_id
        }), 201
    
    except Exception as e:
        connection.close()
        return jsonify({'error': str(e)}), 500

@app.route('/api/all_translations', methods=['GET'])
def get_all_translations():
    """API для получения всех переводов"""
    source_lang = request.args.get('source_lang', '')
    target_lang = request.args.get('target_lang', '')
    
    connection = get_database()
    cursor = connection.cursor()
    
    if source_lang and target_lang:
        cursor.execute('''
            SELECT word_source, word_target FROM translation
            WHERE language_source = ? AND language_target = ?
            ORDER BY word_source
        ''', (source_lang, target_lang))
    else:
        cursor.execute('''
            SELECT word_source, language_source, word_target, language_target FROM translation
            ORDER BY word_source
        ''')
    
    results = cursor.fetchall()
    connection.close()
    
    translations = [dict(row) for row in results]
    return jsonify({'translations': translations})

@app.route('/api/delete_translation/<int:translation_id>', methods=['DELETE'])
def delete_translation(translation_id):
    """API для удаления перевода по ID"""
    connection = get_database()
    cursor = connection.cursor()
    
    cursor.execute('DELETE FROM translation WHERE id = ?', (translation_id,))
    connection.commit()
    
    if cursor.rowcount == 0:
        connection.close()
        return jsonify({'error': 'Перевод не найден'}), 404
    
    connection.close()
    return jsonify({'message': 'Перевод успешно удален'}), 200
```

@app.route('/api/statistics', methods=['GET'])
def get_statistics():
    """API для получения статистики словаря"""
    connection = get_database()
    cursor = connection.cursor()
    
    # Общее количество переводов
    cursor.execute('SELECT COUNT(*) as count FROM translation')
    total_count = cursor.fetchone()['count']
    
    # Количество уникальных исходных слов
    cursor.execute('SELECT COUNT(DISTINCT word_source) as count FROM translation WHERE language_source = "English"')
    unique_english = cursor.fetchone()['count']
    
    # Количество уникальных целевых слов
    cursor.execute('SELECT COUNT(DISTINCT word_target) as count FROM translation WHERE language_target = "Russian"')
    unique_russian = cursor.fetchone()['count']
    
    # Слова с несколькими переводами
    cursor.execute('''
        SELECT word_source, COUNT(*) as translation_count
        FROM translation
        WHERE language_source = "English"
        GROUP BY word_source
        HAVING COUNT(*) > 1
        ORDER BY translation_count DESC
    ''')
    
    multiple_translations = [dict(row) for row in cursor.fetchall()]
    connection.close()
    
    return jsonify({
        'total_translations': total_count,
        'unique_source_words': unique_english,
        'unique_target_words': unique_russian,
        'words_with_multiple_translations': multiple_translations
    })

@app.route('/api/search_by_pattern', methods=['GET'])
def search_by_pattern():
    """API для поиска слов по шаблону (wildcard)"""
    pattern = request.args.get('pattern', '')
    language = request.args.get('language', 'English')
    
    if len(pattern) < 2:
        return jsonify({'error': 'Шаблон должен содержать минимум 2 символа'}), 400
    
    connection = get_database()
    cursor = connection.cursor()
    
    # Заменить * на % для SQL LIKE
    sql_pattern = pattern.replace('*', '%')
    
    if language == 'English':
        cursor.execute('''
            SELECT DISTINCT word_source FROM translation
            WHERE language_source = "English"
            AND word_source LIKE ?
            ORDER BY word_source
        ''', (sql_pattern,))
    else:
        cursor.execute('''
            SELECT DISTINCT word_target FROM translation
            WHERE language_target = "Russian"
            AND word_target LIKE ?
            ORDER BY word_target
        ''', (sql_pattern,))
    
    results = [row[0] for row in cursor.fetchall()]
    connection.close()
    
    return jsonify({'results': results})
```

### Задача 1: Интеграция с реальным переводчиком

Используйте бесплатное API MyMemory Translation API для автоматического заполнения переводов:

```python
import requests

def get_automatic_translation(word, source_lang='en', target_lang='ru'):
    """Получить перевод с помощью MyMemory API"""
    url = f'https://api.mymemory.translated.net/get'
    params = {
        'q': word,
        'langpair': f'{source_lang}|{target_lang}'
    }
    
    response = requests.get(url, params=params)
    data = response.json()
    
    if data['responseStatus'] == 200:
        return data['responseData']['translatedText']
    return None

# Использование
translation = get_automatic_translation('hello', 'en', 'ru')
print(translation)  # Output: привет
```
