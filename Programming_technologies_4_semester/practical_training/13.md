## Введение

В этом практическом уроке мы разработаем веб-приложение, которое позволяет выполнять поиск слов и выражений в текстовом корпусе. Это приложение будет использовать фреймворк Flask для создания веб-интерфейса и базу данных SQLite для хранения текстов.

**Цели работы:**

1. Понимание архитектуры веб-приложений на Flask
2. Работа с базами данных SQLite в Python
3. Реализация функций поиска по текстам
4. Создание пользовательского интерфейса для лингвистического анализа

## Теоретический материал

### Основные компоненты приложения

#### Flask

Flask — это микрофреймворк для разработки веб-приложений на Python. Основные компоненты:

1. **Роуты (Routes)** — функции, которые обрабатывают запросы на определённые URL-адреса
2. **Шаблоны (Templates)** — файлы HTML с поддержкой Jinja2 для динамической генерации содержимого
3. **Статические файлы** — CSS, JavaScript, изображения

#### SQLite

SQLite — это встроённая база данных, которая хранит данные в одном файле. Для работы используется стандартный модуль `sqlite3`.

#### Поиск в текстах

Для эффективного поиска используются следующие техники:

1. **Полнотекстовый поиск** — поиск по всему содержимому
2. **Частичное совпадение** — поиск с использованием оператора LIKE
3. **Контекст слова** — поиск с окружающими словами

def search_texts(query):
    connection = sqlite3.connect('corpus.db')
    cursor = connection.cursor()
    
    # Использование LIKE для частичного совпадения
    cursor.execute('''
        SELECT id, title, content FROM texts

        ORDER BY id

    results = cursor.fetchall()
    connection.close()
    return results

### Шаг 2: Создание модуля базы данных

Создайте файл `database.py`:

```python
import sqlite3
from datetime import datetime

DATABASE = 'corpus.db'

def init_database():
    """Инициализирует базу данных и создаёт таблицу текстов."""
    connection = sqlite3.connect(DATABASE)
    cursor = connection.cursor()
    
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS texts (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            title TEXT NOT NULL,
            content TEXT NOT NULL,
            author TEXT,
            language TEXT DEFAULT 'unknown',
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
        )
    ''')
    
    connection.commit()
    connection.close()


def add_text(title, content, author, language='unknown'):
    """Добавляет новый текст в базу данных."""
    try:
        connection = sqlite3.connect(DATABASE)
        cursor = connection.cursor()
        
        cursor.execute('''
            INSERT INTO texts (title, content, author, language)
            VALUES (?, ?, ?, ?)
        ''', (title, content, author, language))
        
        connection.commit()
        connection.close()
        return True
    except Exception as e:
        print(f"Error adding text: {e}")
        return False


def search_by_word(query):
    """Ищет слово или фразу в текстах."""
    try:
        connection = sqlite3.connect(DATABASE)
        cursor = connection.cursor()
        
        cursor.execute('''
            SELECT id, title, content, author FROM texts
            WHERE content LIKE ?
            ORDER BY id
        ''', (f'%{query}%',))
        
        results = cursor.fetchall()
        connection.close()
        return results
    except Exception as e:
        print(f"Error searching: {e}")
        return []


def get_context(text_id, word, context_size=5):
    """Извлекает контекст слова в тексте."""
    try:
        connection = sqlite3.connect(DATABASE)
        cursor = connection.cursor()
        
        cursor.execute('''
            SELECT content FROM texts WHERE id = ?
        ''', (text_id,))
        
        result = cursor.fetchone()
        connection.close()
        
        if not result:
            return None
        
        content = result[0]
        words = content.split()
        
        # Поиск позиции слова
        positions = []
        for i, w in enumerate(words):
            if word.lower() in w.lower():
                positions.append(i)
        
        # Формирование контекста
        contexts = []
        for pos in positions:
            start = max(0, pos - context_size)
            end = min(len(words), pos + context_size + 1)
            context = ' '.join(words[start:end])
            contexts.append(context)
        
        return contexts
    except Exception as e:
        print(f"Error getting context: {e}")
        return None


def get_all_texts():
    """Получает все тексты из базы данных."""
    try:
        connection = sqlite3.connect(DATABASE)
        cursor = connection.cursor()
        
        cursor.execute('''
            SELECT id, title, author FROM texts
            ORDER BY created_at DESC
        ''')
        
        results = cursor.fetchall()
        connection.close()
        return results
    except Exception as e:
        print(f"Error getting texts: {e}")
        return []


def delete_text(text_id):
    """Удаляет текст из базы данных."""
    try:
        connection = sqlite3.connect(DATABASE)
        cursor = connection.cursor()
        
        cursor.execute('DELETE FROM texts WHERE id = ?', (text_id,))
        
        connection.commit()
        connection.close()
        return True
    except Exception as e:
        print(f"Error deleting text: {e}")
        return False


def count_word_occurrences(word):
    """Подсчитывает количество вхождений слова."""
    try:
        connection = sqlite3.connect(DATABASE)
        cursor = connection.cursor()
        
        cursor.execute('''
            SELECT COUNT(*) FROM texts WHERE content LIKE ?
        ''', (f'%{word}%',))
        
        count = cursor.fetchone()[0]
        connection.close()
        return count
    except Exception as e:
        print(f"Error counting occurrences: {e}")
        return 0
```

### Шаг 3: Создание основного приложения Flask

Создайте файл `app.py`:

```python
from flask import Flask, render_template, request, redirect, url_for, jsonify
from database import (
    init_database, add_text, search_by_word, get_context,
    get_all_texts, delete_text, count_word_occurrences
)

app = Flask(__name__)

@app.before_first_request
def initialize():
    """Инициализирует базу данных при первом запуске приложения."""
    init_database()


@app.route('/')
def index():
    """Главная страница приложения."""
    texts = get_all_texts()
    return render_template('index.html', texts=texts)


@app.route('/add', methods=['GET', 'POST'])
def add():
    """Страница добавления нового текста."""
    if request.method == 'POST':
        title = request.form.get('title')
        content = request.form.get('content')
        author = request.form.get('author', 'Unknown')
        language = request.form.get('language', 'unknown')
        
        if title and content:
            if add_text(title, content, author, language):
                return redirect(url_for('index'))
    
    return render_template('add_text.html')


@app.route('/search', methods=['GET', 'POST'])
def search():
    """Страница поиска по текстам."""
    results = []
    query = ''
    
    if request.method == 'POST':
        query = request.form.get('query', '').strip()
        
        if query:
            results = search_by_word(query)
    
    return render_template('search.html', results=results, query=query)


@app.route('/context/<int:text_id>', methods=['GET'])
def get_text_context(text_id):
    """API для получения контекста слова в тексте."""
    word = request.args.get('word', '')
    context_size = request.args.get('context_size', 5, type=int)
    
    if word:
        contexts = get_context(text_id, word, context_size)
        if contexts:
            return jsonify({
                'success': True,
                'word': word,
                'contexts': contexts
            })
    
    return jsonify({'success': False, 'message': 'Word not found'})


@app.route('/statistics/<word>')
def word_statistics(word):
    """Страница со статистикой по слову."""
    count = count_word_occurrences(word)
    results = search_by_word(word)
    
    return render_template('statistics.html', 
                         word=word, 
                         count=count, 
                         results=results)


@app.route('/delete/<int:text_id>', methods=['POST'])
def delete(text_id):
    """Удаляет текст из базы данных."""
    if delete_text(text_id):
        return redirect(url_for('index'))
    
    return redirect(url_for('index'))


if __name__ == '__main__':
    app.run(debug=True)
```

def search_statistics(query):
    """Получает статистику по слову."""
    try:
        connection = sqlite3.connect(DATABASE)
        cursor = connection.cursor()
        
        # Количество текстов со словом
        cursor.execute('''
            SELECT COUNT(DISTINCT id) FROM texts
            WHERE content LIKE ?
        ''', (f'%{query}%',))
        
        texts_count = cursor.fetchone()[0]
        
        # Общее количество текстов
        cursor.execute('SELECT COUNT(*) FROM texts')
        total_texts = cursor.fetchone()[0]
        
        connection.close()
        
        percentage = (texts_count / total_texts * 100) if total_texts > 0 else 0
        
        return {
            'texts_with_word': texts_count,
            'total_texts': total_texts,
            'percentage': round(percentage, 2)
        }
    except Exception as e:
        print(f"Error getting statistics: {e}")
        return None

import pymystem3

mystem = pymystem3.Mystem()

def search_with_morphology(word):
    """Ищет все формы слова в текстах."""
    try:
        # Получение основной формы слова
        lemmas = mystem.lemmatize(word)
        base_form = lemmas[0] if lemmas else word
        
        connection = sqlite3.connect(DATABASE)
        cursor = connection.cursor()
        
        cursor.execute('''
            SELECT id, title, content, author FROM texts
            ORDER BY id
        ''')
        
        results = cursor.fetchall()
        connection.close()
        
        # Фильтрация результатов
        filtered_results = []
        for result in results:
            text_lemmas = mystem.lemmatize(result[2].lower())
            if base_form in text_lemmas:
                filtered_results.append(result)
        
        return filtered_results
    except Exception as e:
        print(f"Error in morphological search: {e}")
        return []
